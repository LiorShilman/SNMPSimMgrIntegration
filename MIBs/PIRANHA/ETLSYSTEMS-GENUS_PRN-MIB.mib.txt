			dCInjVoutSet 				Integer32,

		MAX-ACCESS read-write
		STATUS current

			the low limit alarm threshold. If the current is less than this limit 
			an alarm will be raised.
			To disable alarm triggering on low limit set value to -9999.
			Consult module specification for permissable limit range as this may 
			differ between module types."
		::= { prnxxxSettingsEntry 4 }
		
	dCInjCurrentLimitHigh OBJECT-TYPE
		SYNTAX Integer32
		MAX-ACCESS read-write
		STATUS current
		DESCRIPTION
			"DC injector current high limit in mA, i.e. set to 250 for 0.25 amps as
			the high limit alarm threshold. If the current is greater than this limit 
			an alarm will be raised.
			To disable alarm triggering on high limit set value to 9999.
			Consult module specification for permissable limit range as this may 
			differ between module types."
		::= { prnxxxSettingsEntry 5 }
		
	rFPowerLimitLow OBJECT-TYPE
		SYNTAX Integer32
		MAX-ACCESS read-write
		STATUS current
		DESCRIPTION
			"RF power detector low limit in hundreths of dBm, i.e. set to -2000 for -20dBm as
			the low limit alarm threshold. If the power is less than this limit 
			an alarm will be raised.
			To disable alarm triggering on low limit set value to -99.
			Consult module specification for permissable limit range as this may 
			differ between module types."
		::= { prnxxxSettingsEntry 6 }
		
	rFPowerLimitHigh OBJECT-TYPE
		SYNTAX Integer32
		MAX-ACCESS read-write
		STATUS current
		DESCRIPTION
			"RF power detector high limit in hundreths of dBm, i.e. set to 500 for +5dBm as

			an alarm will be raised.
			To disable alarm triggering on high limit set value to 99.
			Consult module specification for permissable limit range as this may 
			differ between module types."
		::= { prnxxxSettingsEntry 7 }
